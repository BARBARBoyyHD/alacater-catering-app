## 1. Executive Summary

### 1.1 Purpose

This document provides the technical architecture and implementation guidelines for Alacater. It follows **DRY** and **KISS** principles by referencing detailed specifications instead of duplicating them.

**For a high-level overview, start with:** [OpenSpec](../openspec.md)

### 1.2 Architecture Principles

| Principle | Description |
|-----------|-------------|
| **Serverless-First** | Supabase + Vercel for minimal infrastructure |
| **Type Safety** | End-to-end TypeScript with Prisma |
| **Component-Driven** | Reusable UI components ([Color Guidelines](./02.Color-Guidelines.md)) |
| **Mobile-First** | React Native (customers), Next.js (providers) |
| **Security by Design** | RLS, JWT, encryption at every layer |

## 4. Database Schema (Prisma)

### 4.1 Complete Prisma Schema

```prisma
// database/schema.prisma

generator client {
  provider = "prisma-client-js"
  previewFeatures = ["multiSchema", "deno"]
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
  directUrl = env("DIRECT_URL")  // Required for Supabase connection pooling
}

// ============================================
// USER & AUTHENTICATION
// ============================================

model User {
  id                String    @id @default(uuid())
  email             String    @unique
  phone             String?   @unique
  fullName          String?
  avatarUrl         String?
  role              UserRole  @default(CUSTOMER)
  emailVerified     Boolean   @default(false)
  phoneVerified     Boolean   @default(false)
  createdAt         DateTime  @default(now())
  updatedAt         DateTime  @updatedAt
  lastLoginAt       DateTime?

  // Relations
  customerProfile   Customer?
  providerProfile   Provider?
  orders            Order[]
  reviews           Review[]
  addresses         Address[]
  notifications     Notification[]
  referrals         Referral[] @relation("referrer")
  referredBy        Referral?  @relation("referred")

  @@index([email])
  @@index([role])
}

enum UserRole {
  CUSTOMER
  PROVIDER
  ADMIN
}

// ============================================
// CUSTOMER PROFILE
// ============================================

model Customer {
  id                String    @id @default(uuid())
  userId            String    @unique
  user              User      @relation(fields: [userId], references: [id], onDelete: Cascade)

  // Diet Preferences (BRD Section 5.1 - Smart Diet Assistant)
  dietaryPreferences String[] // ["vegan", "keto", "halal", "low-sugar", "no-msg"]
  allergies          String[] // ["peanuts", "shellfish", "dairy"]
  healthGoals        String[] // ["weight-loss", "muscle-gain", "maintenance"]

  // Stats
  totalOrders        Int       @default(0)
  totalSpent         Decimal   @default(0) @db.Decimal(15, 2)
  loyaltyPoints      Int       @default(0)

  createdAt          DateTime  @default(now())
  updatedAt          DateTime  @updatedAt

  // Relations
  orders             Order[]
  reviews            Review[]
  addresses          Address[]

  @@index([userId])
}

// ============================================
// PROVIDER / CATERING PROFILE
// ============================================

model Provider {
  id                String    @id @default(uuid())
  userId            String    @unique
  user              User      @relation(fields: [userId], references: [id], onDelete: Cascade)

  // Business Info
  businessName      String
  slug              String    @unique
  description       String?
  logoUrl           String?
  bannerUrl         String?
  phone             String
  email             String
  website           String?

  // Social Media (BRD Section 5.2 - Branding & Promosi)
  instagramHandle   String?
  facebookHandle    String?

  // Verification (BRD Section 5.2 - Verified Seller Badge)
  isVerified        Boolean   @default(false)
  verifiedAt        DateTime?
  verificationDocs  String[]  // Array of document URLs

  // Subscription Tier (BRD Section 6 - Monetization)
  subscriptionTier  SubscriptionTier @default(FREE)
  subscriptionStart DateTime?
  subscriptionEnd   DateTime?

  // Stats
  totalOrders       Int       @default(0)
  totalRevenue      Decimal   @default(0) @db.Decimal(15, 2)
  averageRating     Decimal   @default(0) @db.Decimal(3, 2)
  totalReviews      Int       @default(0)

  // Settings
  isActive          Boolean   @default(true)
  autoAcceptOrders  Boolean   @default(false)

  createdAt         DateTime  @default(now())
  updatedAt         DateTime  @updatedAt

  // Relations
  products          Product[]
  orders            Order[]
  reviews           Review[]
  deliveryAreas     DeliveryArea[]
  operatingHours    OperatingHours[]
  holidays          ProviderHoliday[]

  @@index([userId])
  @@index([slug])
  @@index([isVerified])
  @@index([subscriptionTier])
}

enum SubscriptionTier {
  FREE
  PREMIUM
}

// ============================================
// PRODUCTS & MENUS
// ============================================

model Product {
  id                String    @id @default(uuid())
  providerId        String
  provider          Provider  @relation(fields: [providerId], references: [id], onDelete: Cascade)

  // Product Info
  name              String
  slug              String
  description       String?
  images            String[]  // Array of image URLs

  // Category
  category          ProductCategory
  dietTags          String[]  // ["vegan", "keto", "halal", "low-calorie"]

  // Pricing
  basePrice         Decimal   @db.Decimal(15, 2)

  // Subscription Options (BRD Section 5.1 - Pemesanan dengan paket langganan)
  durationOptions   DurationOption[]

  // Meal Plan Options
  mealPlanOptions   MealPlanOption[]

  // Additional Menus
  additionalMenus   AdditionalMenu[]

  // Stats
  totalOrders       Int       @default(0)
  averageRating     Decimal   @default(0) @db.Decimal(3, 2)
  totalReviews      Int       @default(0)

  // Status
  isActive          Boolean   @default(true)
  isFeatured        Boolean   @default(false)

  createdAt         DateTime  @default(now())
  updatedAt         DateTime  @updatedAt

  // Relations
  orderItems        OrderItem[]
  reviews           Review[]

  @@index([providerId])
  @@index([category])
  @@index([isActive])
  @@unique([providerId, slug])
}

enum ProductCategory {
  WEIGHT_LOSS
  MUSCLE_GAIN
  KETO
  VEGAN
  HALAL
  BALANCED
  CUSTOM
}

model DurationOption {
  id                String    @id @default(uuid())
  productId         String
  product           Product   @relation(fields: [productId], references: [id], onDelete: Cascade)

  days              Int       // 5, 10, 20, 60 days
  price             Decimal   @db.Decimal(15, 2)
  originalPrice     Decimal?  @db.Decimal(15, 2) // For showing discounts
  savings           Decimal?  @db.Decimal(15, 2)

  isActive          Boolean   @default(true)

  createdAt         DateTime  @default(now())
  updatedAt         DateTime  @updatedAt

  @@index([productId])
}

model MealPlanOption {
  id                String    @id @default(uuid())
  productId         String
  product           Product   @relation(fields: [productId], references: [id], onDelete: Cascade)

  name              String    // "Makan Siang (1 Meal)", "Makan Siang + Malam (2 Meals)"
  mealsPerDay       Int       // 1 or 2
  price             Decimal   @db.Decimal(15, 2)
  originalPrice     Decimal?  @db.Decimal(15, 2)
  savings           Decimal?  @db.Decimal(15, 2)

  isActive          Boolean   @default(true)

  createdAt         DateTime  @default(now())
  updatedAt         DateTime  @updatedAt

  @@index([productId])
}

model AdditionalMenu {
  id                String    @id @default(uuid())
  productId         String
  product           Product   @relation(fields: [productId], references: [id], onDelete: Cascade)

  name              String    // "Snack Bar Diet"
  description       String?
  image             String?
  price             Decimal   @db.Decimal(15, 2)
  unit              String    // "30 hari", "per piece"

  isActive          Boolean   @default(true)

  createdAt         DateTime  @default(now())
  updatedAt         DateTime  @updatedAt

  @@index([productId])
}

// ============================================
// DELIVERY & LOGISTICS
// ============================================

model DeliveryArea {
  id                String    @id @default(uuid())
  providerId        String
  provider          Provider  @relation(fields: [providerId], references: [id], onDelete: Cascade)

  // Area Definition (BRD Section 5.1 - Biaya Pengiriman)
  city              String
  district          String?
  subdistrict       String?

  // Fee Structure
  fee               Decimal   @db.Decimal(15, 2)
  feeType           DeliveryFeeType @default(FLAT)

  // Minimum Order
  minOrder          Decimal?  @db.Decimal(15, 2)

  isActive          Boolean   @default(true)

  createdAt         DateTime  @default(now())
  updatedAt         DateTime  @updatedAt

  @@index([providerId])
  @@index([city])
}

enum DeliveryFeeType {
  FLAT          // Flat fee per order
  PER_KM        // Fee per kilometer
  TIERED        // Tiered pricing based on distance
}

model OperatingHours {
  id                String    @id @default(uuid())
  providerId        String
  provider          Provider  @relation(fields: [providerId], references: [id], onDelete: Cascade)

  dayOfWeek         Int       // 0-6 (Sunday-Saturday)
  startTime         String    // "09:00"
  endTime           String    // "18:00"

  // Meal delivery windows
  lunchStart        String?   // "11:00"
  lunchEnd          String?   // "14:00"
  dinnerStart       String?   // "17:00"
  dinnerEnd         String?   // "20:00"

  isActive          Boolean   @default(true)

  createdAt         DateTime  @default(now())
  updatedAt         DateTime  @updatedAt

  @@index([providerId])
}

model ProviderHoliday {
  id                String    @id @default(uuid())
  providerId        String
  provider          Provider  @relation(fields: [providerId], references: [id], onDelete: Cascade)

  name              String    // "Libur Nasional", "Cuti Bersama"
  startDate         DateTime  @db.Date
  endDate           DateTime  @db.Date

  // Recurring yearly
  isRecurring       Boolean   @default(false)

  createdAt         DateTime  @default(now())
  updatedAt         DateTime  @updatedAt

  @@index([providerId])
}

// ============================================
// ORDERS & TRANSACTIONS
// ============================================

model Order {
  id                String    @id @default(uuid())
  orderNumber       String    @unique // Format: ORD-YYYYMMDD-XXXX

  // Parties
  customerId        String
  customer          Customer  @relation(fields: [customerId], references: [id])
  providerId        String
  provider          Provider  @relation(fields: [providerId], references: [id])

  // Product
  productId         String
  product           Product   @relation(fields: [productId], references: [id])

  // Order Configuration
  durationOptionId  String?
  durationOption    DurationOption? @relation(fields: [durationOptionId], references: [id])

  mealPlanOptionId  String?
  mealPlanOption    MealPlanOption? @relation(fields: [mealPlanOptionId], references: [id])

  // Additional Items
  additionalMenus   OrderAdditionalMenu[]

  // Quantity
  quantity          Int       @default(1)

  // Pricing (BRD Section 6 - Monetization)
  subtotal          Decimal   @db.Decimal(15, 2)
  deliveryFee       Decimal   @db.Decimal(15, 2)
  platformFee       Decimal   @default(5000) @db.Decimal(15, 2) // Flat Rp5,000
  discount          Decimal   @default(0) @db.Decimal(15, 2)
  discountCode      String?
```

### 6.2 Row Level Security (RLS) Policies

```sql
-- Supabase RLS Policies

-- Users table
ALTER TABLE users ENABLE ROW LEVEL SECURITY;

-- Users can read their own data
CREATE POLICY "Users can view own data"
  ON users FOR SELECT
  USING (auth.uid() = id);

-- Users can update their own data
CREATE POLICY "Users can update own data"
  ON users FOR UPDATE
  USING (auth.uid() = id);

-- Providers table
ALTER TABLE providers ENABLE ROW LEVEL SECURITY;

-- Anyone can read active providers
CREATE POLICY "Anyone can view active providers"
  ON providers FOR SELECT
  USING (is_active = true);

-- Providers can update their own data
CREATE POLICY "Providers can update own data"
  ON providers FOR UPDATE
  USING (auth.uid() = user_id);

-- Orders table
ALTER TABLE orders ENABLE ROW LEVEL SECURITY;

-- Customers can view their own orders
CREATE POLICY "Customers can view own orders"
  ON orders FOR SELECT
  USING (auth.uid() = (
    SELECT user_id FROM customers WHERE id = orders.customer_id
  ));

-- Providers can view orders for their products
CREATE POLICY "Providers can view own orders"
  ON orders FOR SELECT
  USING (auth.uid() = (
    SELECT user_id FROM providers WHERE id = orders.provider_id
  ));

-- Reviews table
ALTER TABLE reviews ENABLE ROW LEVEL SECURITY;

-- Anyone can read visible reviews
CREATE POLICY "Anyone can view reviews"
  ON reviews FOR SELECT
  USING (is_visible = true);

-- Only order customer can create review
CREATE POLICY "Customers can create reviews for own orders"
  ON reviews FOR INSERT
  WITH CHECK (auth.uid() = (
    SELECT user_id FROM customers WHERE id = reviews.customer_id
  ));
```

## 12. Security Considerations

### 12.1 Security Checklist

| Area                   | Implementation                               |
| ---------------------- | -------------------------------------------- |
| **Authentication**     | Supabase Auth with JWT, refresh tokens       |
| **Authorization**      | Row Level Security (RLS) on all tables       |
| **API Security**       | Rate limiting via Vercel middleware          |
| **Data Encryption**    | TLS 1.3 in transit, AES-256 at rest          |
| **Payment Security**   | PCI DSS compliant via Midtrans/Xendit        |
| **Input Validation**   | Zod schemas for all inputs                   |
| **XSS Prevention**     | React escapes by default, CSP headers        |
| **CSRF Prevention**    | SameSite cookies, CSRF tokens                |
| **Secrets Management** | GitHub Secrets, Vercel Environment Variables |

### 12.2 Supabase Security Setup

```sql
-- Enable required extensions
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";
CREATE EXTENSION IF NOT EXISTS "pgcrypto";

-- Audit logging trigger
CREATE OR REPLACE FUNCTION audit_changes()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO activity_logs (
    entity_type,
    entity_id,
    action,
    changes,
    created_at
  ) VALUES (
    TG_TABLE_NAME,
    COALESCE(NEW.id, OLD.id),
    TG_OP,
    row_to_json(NEW)::jsonb - row_to_json(OLD)::jsonb,
    NOW()
  );
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
```
